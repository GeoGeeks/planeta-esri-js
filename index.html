Mapa 2D

                    

// Cargar módulos necesarios
require([
    "esri/Map",
    "esri/views/MapView",
], function(Map, MapView) {

    // Crear mapa
    const map = new Map ({
        basemap: "topo"
    });

    // Crear vista
    const mapView = new MapView ({
        container: "map1",  \\ Contenedor HTML
        map: map,
    });
});

Agregar FeatureLayer

                    

// Cargar módulos necesarios
require([
    "esri/Map",
    "esri/views/MapView",
    "esri/layers/FeatureLayer",
], function(Map, MapView, FeatureLayer) {

    // Cargar capa
    const generationPlants = new FeatureLayer ({
        url: "http://arcgis.simec.gov.co:6080/arcgis/rest/services/UPME_EN/UPME_EN_Generaci%C3%B3n_Real/MapServer/0"
    });

    // Crear mapa
    const map = new Map ({
        basemap: "dark-gray",
        layers = [generationPlants]  // agregar capa
    });
});

Agregar TileLayer y cambiar simbología

                    

const basemapFirefly = new TileLayer ({
    url: "http://serviceslab.arcgisonline.com/arcgis/rest/services/Firefly_World_Imagery/MapServer"
});

generationPlants.renderer = {
    type: "simple",
    symbol: {
        type: "picture-marker",
        url: "https://static.arcgis.com/images/Symbols/Firefly/FireflyB8.png",
        width: 16,
        height: 16
    }
};

const map4 = new Map ({
    layers: [basemapFirefly, generationPlants]
});

Agregar leyenda

                    

const legend = new Expand({
    content:new Legend ({
        view: mapView,
        style: "card",
        layerInfos: [{
            title: "Plantas de Generación",
            layer: generationPlants
        }]
    }),
    view: mapView
});

mapView.ui.add(legend, "bottom-right");
